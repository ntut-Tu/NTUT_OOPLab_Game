14class MenuScene : public Scene {
15public:
16 enum class SceneMode { DEFAULT, MAP, MENU, TUTORIAL, SANDBOX };
17
18public:
19 MenuScene()
20 : m_BGM("../assets/BGM/RA3-Soviet_March.mp3") {}
21 ~MenuScene() {}
22
23 void Start() override {
24 m_MapScene->Start();
25 m_DefaultScene->Start();
26 m_TutorialScene->Start();
27 m_SandBoxScene->Start();
28 }
29
30 void Update() override {
31 if (m_CurrentMode == SceneMode::MENU) {
32 if (Util::Input::IsKeyPressed(Util::Keycode::P)) {
33 m_BGM.Play();
34 }
35 if (Util::Input::IsKeyPressed(Util::Keycode::K)) {
36 m_BGM.LoadMedia("../assets/BGM/Jimmy_Scott-Sycamore_Trees.mp3");
37 m_BGM.Play();
38 }
39 if (Util::Input::IsKeyPressed(Util::Keycode::M)) {
40 m_CurrentMode = SceneMode::MAP;
41 }
42 if (Util::Input::IsKeyPressed(Util::Keycode::D)) {
43 m_CurrentMode = SceneMode::DEFAULT;
44 }
45 if (Util::Input::IsKeyPressed(Util::Keycode::T) ||
46 m_MenuUI.getUIStatus()->getUIStatusType() ==
47 UIStatusType::UI_START) {
48 m_MenuUI.getUIStatus()->setUIStatusType(
49 UIStatusType::UI_DEFAULT);
50 m_CurrentMode = SceneMode::TUTORIAL;
51 }
52 if (Util::Input::IsKeyPressed(Util::Keycode::S)) {
53 m_CurrentMode = SceneMode::SANDBOX;
54 }
55 }
56
57 switch (m_CurrentMode) {
58 case (SceneMode::MAP):
59 m_MapScene->Update();
60 break;
61 case (SceneMode::DEFAULT):
62 m_DefaultScene->Update();
63 break;
64 case (SceneMode::TUTORIAL):
65 m_TutorialScene->Update();
66 break;
67 case (SceneMode::SANDBOX):
68 m_SandBoxScene->Update();
69 break;
70 case (SceneMode::MENU):
71 m_MenuUI.Update();
72 break;
73 }
74 }
75
76 SceneMode getSceneMode() { return m_CurrentMode; }
77
78 bool ifExit() {
79 if (m_MenuUI.getUIStatus()->getUIStatusType() ==
80 UIStatusType::UI_EXIT) {
81 return true;
82 }
83 return false;
84 }
85
86private:
87 SceneMode m_CurrentMode = SceneMode::MENU;
88 std::shared_ptr<TutorialScene> m_TutorialScene =
89 std::make_shared<TutorialScene>();
90 std::shared_ptr<MapScene> m_MapScene = std::make_shared<MapScene>();
91 std::shared_ptr<DefaultScene> m_DefaultScene =
92 std::make_shared<DefaultScene>();
93 std::shared_ptr<SandBoxScene> m_SandBoxScene =
94 std::make_shared<SandBoxScene>();
95 Util::BGM m_BGM;
96
97 MenuUI m_MenuUI;
98};
