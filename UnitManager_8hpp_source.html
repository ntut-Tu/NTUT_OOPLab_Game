117 void spawn(std::shared_ptr<MapClass> m_Map,UnitType unit,HouseType house,glm::vec2 cellPos){
118 cellPos = MapUtil::CellCoordToGlobal(cellPos);
119 //缺檢查敵方擁有建築的位置，並重生在該處
120 if(house==HouseType::ENEMY){
121 m_Enemy->addUnitConstructCount(unit, 1);
122 }else{
123 // m_Player->setUnitConstructCount(unit, 1);
124 }
125 switch (unit) {
126 case UnitType::BARRACKS: {
127 auto structure = std::make_shared<Barracks>(house);
128 structure->Start(cellPos);
129 m_StructureManager->getStructureArray()->buildNewStructure(m_Map,structure, true);
130 break;
131 }
132 case UnitType::ORE_REF: {
133 auto structure = std::make_shared<OreRefinery>(house);
134 structure->Start(cellPos);
135 m_StructureManager->getStructureArray()->buildNewStructure(m_Map,structure, true);
136 break;
137 }
138 case UnitType::POWER_PLANT: {
139 auto structure = std::make_shared<PowerPlants>(house);
140 structure->Start(cellPos);
141 m_StructureManager->getStructureArray()->buildNewStructure(m_Map,structure, true);
142 break;
143 }
144 case UnitType::WAR_FACT: {
145 auto structure = std::make_shared<WarFactory>(house);
146 structure->Start(cellPos);
147 m_StructureManager->getStructureArray()->buildNewStructure(m_Map,structure, true);
148 break;
149 }
150 case UnitType::ADV_POWER_PLANT: {
151 auto structure = std::make_shared<ADVPowerPlants>(house);
152 structure->Start(cellPos);
153 m_StructureManager->getStructureArray()->buildNewStructure(m_Map,structure, true);
154 break;
155 }
156 case UnitType::INFANTRY:{
157 auto avatar = std::make_shared<Infantry>(house);
158 avatar ->Start(cellPos);
159// avatar ->setNewDestination(cellPos);
160 m_AvatarManager->AppendAvatar(avatar);
161 break;
162 }
163 case UnitType::NONE: {
164 printf("(GOM)error! try to build when type == NONE\n");
165 break;
166 }
167 default: {
168 printf("(GOM)error! try to spawn unknown type\n");
169 break;
170 }
171 }
