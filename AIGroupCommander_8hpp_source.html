9#define GROUP_SIZE 4
10#define AUTO_FIND_RANGE 4
11#define AUTO_ATTACK_METHOD 2
12class AIGroupCommander {
13private:
14 std::shared_ptr<UnitManager> m_PlayerUnitManager;
15 std::shared_ptr<UnitManager> m_AIUnitManager;
16 std::shared_ptr<AvatarManager> m_AIAvatarManager;
17 std::shared_ptr<MapClass> m_Map;
18
19 std::vector<std::vector<std::shared_ptr<Avatar>>> m_offensiveGroup;
20 std::vector<std::shared_ptr<Avatar>> m_defensiveGroup;
21public:
22 AIGroupCommander(std::shared_ptr<UnitManager> PlayerUnitManager,std::shared_ptr<UnitManager> AIUnitManager,std::shared_ptr<MapClass> Map):m_PlayerUnitManager(PlayerUnitManager),m_AIUnitManager(AIUnitManager),m_Map(Map){};
23 ~AIGroupCommander(){};
24 void Start(){
25 m_AIAvatarManager = m_AIUnitManager->getAvatarManager();

34 void setAllTroopToAttackMode(){
35 if(m_defensiveGroup.empty()){
36 return;
37 }
38 for(auto i : m_defensiveGroup){
39 addTroopToOffensiveGroup(i);
40 }
41 m_defensiveGroup.clear();

43 void setTroopToAttackMode(int num){
44 if(m_defensiveGroup.empty()){
45 return;
46 }
47 int max = static_cast<int>(m_defensiveGroup.size());
48 for(int i=max-1;i>=0 && num>=max-i;i--){
49 addTroopToOffensiveGroup(m_defensiveGroup[i]);
50 m_defensiveGroup.pop_back();
51 }
52 }

58 int getOffensiveTroopSize(){
59 int totalSize = 0;
60 for (const auto& row : m_offensiveGroup) {
61 totalSize += static_cast<int>(row.size());

65protected:
66 void updateOffensiveGroup(){
67 for(auto i : m_offensiveGroup){
68 for(int j=0;j<static_cast<int>(i.size());j++){
69 if(i[j]->getHealth()->ifDead()){
70 i.erase(i.begin()+j);
71 }
72 }
73 }
74 updateOffensiveTroopAttackTarget();
75 }
76 void updateDefensiveGroup(){
77 for(int i = static_cast<int>(m_defensiveGroup.size());i>0;--i ){
78 if(m_defensiveGroup[i-1]->getHealth()->ifDead()){
79 m_defensiveGroup.erase(m_defensiveGroup.begin()+i-1);
80 }else{
81 autoAttack(m_defensiveGroup[i-1],AUTO_ATTACK_METHOD);
82 };
83 }
84 std::vector<std::shared_ptr<Avatar>> temp = m_AIAvatarManager->getAvatarArray();
85 for(int i=static_cast<int>(temp.size())-1;i>=0;i--){
86 if(temp[i]->getAIType()==AI_Type::NONE){
87 temp[i]->setAIType(AI_Type::DEFENCE);
88 m_defensiveGroup.push_back(temp[i]);
89 }
90 }
91 }
92 void addTroopToOffensiveGroup(std::shared_ptr<Avatar> unit){
93 auto it = std::find_if(m_offensiveGroup.begin(), m_offensiveGroup.end(), [](const std::vector<std::shared_ptr<Avatar>>& group) {
94 return group.size() < GROUP_SIZE;
95 });
96 unit->setAIType(AI_Type::ATTACK);
97 if (it != m_offensiveGroup.end()) {
98 it->push_back(unit);
99 } else {
100 m_offensiveGroup.push_back(std::vector<std::shared_ptr<Avatar>>());
101 m_offensiveGroup.back().push_back(unit);
102 }
103 }
104 void autoAttack(std::shared_ptr<Avatar> unit,int method){
105 switch (method) {
106 case 1:
107 for(auto i : m_PlayerUnitManager->getAvatarManager()->getAvatarArray()){
108 if(i->getDistance(unit->getCurrentLocationInCell())<=AUTO_FIND_RANGE){
109 //attack
110 m_AIAvatarManager->assignAttackOrderToAvatar(unit,i->getCurrentLocationInCell(),HouseType::ENEMY);
111 return;
112 }
113 }
114 break;
115 case 2:
116 glm::vec2 targetCell = m_Map->findEnemyInRange(AUTO_FIND_RANGE,unit->getCurrentLocationInCell(),HouseType::ENEMY);
117 if(targetCell.x == -1.f && targetCell.y == -1.f){
118 return;
119 }
120 //attack
121 m_AIAvatarManager->assignAttackOrderToAvatar(unit,targetCell,HouseType::ENEMY);
122 break;
123 }
124 }
125 void updateOffensiveTroopAttackTarget(){
126 for(int i = static_cast<int>(m_offensiveGroup.size());i>0;--i){
127 auto& currentGroup = m_offensiveGroup[i-1];
128 printf("(updateOffensiveTroopAttackTarget)groupSize : %d\n",static_cast<int>(currentGroup.size()));
129 if(!currentGroup.empty()){
130 if (!m_AIAvatarManager->ifAvatarHasNemesis(currentGroup.front())){
131 // find new target
132 glm::vec2 targetCell = findTargetForOffensiveUnit(currentGroup.front());
133 if(targetCell.x<0){
134 printf("(updateOffensiveTroopAttackTarget) No Target\n");
135 break;
136 }
137 // attack
138 for (int j=static_cast<int>(currentGroup.size());j>0;--j) {
139 if(currentGroup[j-1]->getHealth()->ifDead()){
140 currentGroup.erase(currentGroup.begin()+j-1);
141 }else{
142 m_AIAvatarManager->assignAttackOrderToAvatar(
143 currentGroup[j-1], targetCell,HouseType::ENEMY);
144 }
145 }
146 }else{
147 glm::vec2 targetCell = m_AIAvatarManager->getAvatarNemesisCell(currentGroup.front());
148 printf("(updateOffensiveTroopAttackTarget)AvatarHasNemesis : {%d,%d}\n",static_cast<int>(targetCell.x),static_cast<int>(targetCell.y));
149 }
150 }else{
151 m_offensiveGroup.erase(m_offensiveGroup.begin()+i-1);
152 }
153 }
154 }

156 glm::vec2 findTargetForOffensiveUnit(std::shared_ptr<Avatar> unit){
157 glm::vec2 targetCell = {-10.f,-10.f};
158 //issue: empty group should delete before this
159 if(static_cast<int>(m_offensiveGroup.size())>=static_cast<int>(m_PlayerUnitManager->getAvatarManager()->getAvatarArray().size())){
160 if(!m_PlayerUnitManager->getStructureManager()->getStructureArray()->getBuiltStructureArray().empty()){
161 targetCell=m_PlayerUnitManager->getStructureManager()->getStructureArray()->getBuiltStructureArray().front()->getTopRightCell();
162 }else{
163 return targetCell;
164 }
165 for(auto i:m_PlayerUnitManager->getStructureManager()->getStructureArray()->getBuiltStructureArray()){
166 if(unit->getDistance(i->getTopRightCell())<unit->getDistance(targetCell)){
167 targetCell = i->getTopRightCell();
168 }
169 }
170 }else{
171 if(!m_PlayerUnitManager->getAvatarManager()->getAvatarArray().empty()){
172 targetCell=m_PlayerUnitManager->getAvatarManager()->getAvatarArray().front()->getCurrentLocationInCell();
173 }
174 for(auto i : m_PlayerUnitManager->getAvatarManager()->getAvatarArray()){
175 if(i->getDistance(unit->getCurrentLocationInCell())<unit->getDistance(targetCell)){
176 //attack
177 if(m_AIAvatarManager->ifAvatarHasNemesis(m_Map->getTileByCellPosition(targetCell)->getAvatars().front())){
178 continue;
179 }
180 targetCell = i->getCurrentLocationInCell();
181 }
182 }
183 }
184 printf("(findTargetForOffensiveUnit) TargetCell : {%d,%d}\n",targetCell.x,targetCell.y);
185 return targetCell;
186 }
